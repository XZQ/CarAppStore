# 04. 类图与关系图

## 1. 7 个模块类图

```mermaid
classDiagram

class AppStoreViewModel {
  +loadHome()
  +loadMyApps()
  +loadAppDetail(appId)
  +clickDownload(appId)
  +clickInstall(appId)
  +clickUpgrade(appId)
  +clickOpen(packageName)
}

class DownloadModule {
  +startDownload(appId)
  +pauseDownload(appId)
  +resumeDownload(appId)
  +cancelDownload(appId)
  +retryDownload(appId)
}

class InstallModule {
  +install(appId, apkPath)
  +retryInstall(appId)
  +queryInstallState(appId)
}

class UpgradeModule {
  +checkUpgrade(appId)
  +startUpgrade(appId)
  +startAllExecutable()
}

class AppManagerModule {
  +getHomeApps()
  +getMyApps()
  +getSearchApps(keyword)
  +getDownloadTasks()
  +getInstallTasks()
  +getUpgradeTasks()
  +getPolicyPrompt()
}

class StateCenter {
  +updateDownloadState(appId, state)
  +updateInstallState(appId, state)
  +updateUpgradeState(appId, state)
  +observeAppState(appId)
  +observeAllAppStates()
}

class PolicyCenter {
  +canDownload(appId)
  +canInstall(appId)
  +canUpgrade(appId)
  +setNetworkState(state)
  +setVehicleState(state)
  +setStorageState(state)
}

class AppRepository {
  +fetchHomeApps()
  +fetchAppDetail(appId)
  +fetchUpgradeInfo(appId)
  +queryInstalledApps()
  +saveDownloadTask(record)
  +loadDownloadTasks()
}

class FileDownloader {
  <<interface>>
  +download(task)
  +pause(taskId)
  +resume(taskId)
  +cancel(taskId)
}

class PackageInstaller {
  <<interface>>
  +install(apkPath)
}
```
